docker run -it ubuntu:20.04 /bin/bash
//跑ubuntu

docker ps
//罗列出所有在跑的镜像容器

exit;
//退出该镜像

ctrl+p+q
//run进去容器后,但是退出容器不停止

docker stop (容器名字)
//停止容器

docker kill (容器名字)
//强制停止

docker start (容器名字)
//重新启动容器

docker rmi (容器) (i表示image)
//删除镜像

docker rm (容器)
//删除已经停止的容器

docker rm -f (容器名字)
//强制删除

docker rm -f $(docker ps -a -q)
//强制删除所有容器
